### Brute Force
```py
def bruteForce(text, pattern):
    n = len(text)
    m = len(pattern)
    i = j = 0
    while i < n and j < m:
        if text[i] == pattern[j]:
            i += 1
            j += 1
        else:
            i = i - j + 1
            j = 0
    if j == m:
        return i - m
    return -1

a = 'pineapple'
b = 'apple'
print(bruteForce(a, b))  # => 4
```
